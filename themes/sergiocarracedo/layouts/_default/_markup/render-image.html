{{- $image := .Page.Resources.GetMatch .Destination -}} {{- /* The following is is used as a fallback if the image is not found or if it's a GIF. */ -}} {{- $default := printf "" (.Destination | safeURL) $.PlainText $.Title -}} {{- if $image -}} {{- if or (strings.HasSuffix $image.Name ".gif") (strings.HasSuffix $image.Name ".svg") -}} {{- $default | safeHTML -}} {{- else -}} {{- $imageOriginalWidth := $image.Width -}} {{- $sizes := slice 458 768 1200 -}} {{- $srcset := slice -}} {{- $sizestag := slice -}} {{- range $sizes -}} {{- if lt (mul . 1.2) $imageOriginalWidth -}} {{- $thumb := $image.Resize (printf "%dx webp q90" .) -}} {{- $srcset = $srcset | append (printf ("%s %dw") $thumb.RelPermalink . ) -}} {{- $sizestag = $sizestag | append (printf ("(max-width: %dpx) 100vw") . ) -}} {{- end -}} {{- end -}} {{- if ne (len $srcset) 3 -}} {{- $thumb := $image.Resize (printf "%dx webp q90" $imageOriginalWidth) -}} {{- $srcset = $srcset | append (printf ("%s %dw") $thumb.RelPermalink $imageOriginalWidth ) -}} {{- $sizestag = $sizestag | append (printf ("%dpx") $imageOriginalWidth ) -}} {{- end -}} {{- $imageFallback := $image.Resize "100x jpg q90" -}} {{- end -}} {{- else -}} {{- warnf "Unable to find '%s' -- ensure image exists alongside document in page bundle" .Destination -}} {{- $default | safeHTML -}} {{- end -}}
